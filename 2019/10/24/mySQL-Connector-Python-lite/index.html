class DBOP():

def __init__(self):
    # 使用mysql.connector连接数据库
    self.conn = mysql.connector.connect(
            host="localhost",
            user="root",
            password="123456",
            use_unicode=True
            )

    self.cursor = self.conn.cursor()
    # 若test数据库不存在则创建一个test数据库
    self.cursor.execute("CREATE DATABASE IF NOT EXISTS test")

    self.cursor.execute("USE test")

    # #检查是否当前数据库正确
    # self.cursor.execute("select DATABASE()")
    # self.CurrentDB = self.cursor.fetchone()
    # print("current database is %s" % self.CurrentDB)

    # 若test表不存在则创建一个test表
    self.cursor.execute("CREATE TABLE IF NOT EXISTS test(name VARCHAR(255),num INT(20));")

#增加数据方法
def addOP(self,name,num):
    self.cursor.execute("INSERT INTO test VALUES('%s',%d)" % (name,num))
    self.conn.commit()   # DDL语言需要手动commit
#删除数据方法（可使用单一参数确定name，或者双参数同时确定name和num键匹配的数据）
def delOP(self,name,*num):
    if num == (()):
        self.cursor.execute("DELETE FROM test WHERE name='" + name +"'")
    else:
        self.cursor.execute("DELETE FROM test WHERE name='%s'AND num = %d" % (name,int(num[0])))
    self.conn.commit()

#修改数据方法（可使用单一参数确定name，或者双参数同时确定name和num键匹配的数据）
def alterOP(self,name,anum,*num):
    if num ==(()):
        self.cursor.execute("UPDATE test SET num = %d WHERE name ='%s'"%(anum,name))
    else:
        self.cursor.execute("UPDATE test SET num = %d WHERE name ='%s' AND num = %d"%(anum,name,int(num[0])))
    self.conn.commit()  

#查找数据方法（可使用单一参数确定name，或者双参数同时确定name和num键匹配的数据）
def findOP(self,name,*num):
    if num ==(()):
        self.cursor.execute("SELECT * FROM test WHERE name = '%s'" %name)
    else:
        self.cursor.execute("SELECT * FROM test WHERE name = '%s' AND num = %d" %(name,int(num[0])))
    data = self.cursor.fetchall()
    print(data)
#查找所有的数据
def findALL(self):
    self.cursor.execute("SELECT * FROM test")
    data = self.cursor.fetchall()
    print(data)
